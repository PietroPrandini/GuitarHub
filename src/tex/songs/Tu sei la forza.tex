% Minor:                     \[Am]  \[Bm]  \[Cm]  \[Dm]  \[Em]  \[Fm]  \[Gm]   %

% \rep{n}                 % Repeat n times                                     %

%                             %   with the symbol ^ in the next verses         %
% \beginchorus ... \endchorus % Used for delimiters of a chorus                %
% \ifchorded ... \fi          % Conditional sentences: only if chorded books   %
